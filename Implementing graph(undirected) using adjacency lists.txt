Implementing graph(undirected) using adjacency lists


def addNode(v):
    if v not in k:
        k.append(v)
        matrix[v]=[]
    else:
        print("That node is already available...")
def addEdge(node1,node2,weight):
    if node1 in k:
        if node2 in k:
            l1=[]
            l1.append(node2)
            l1.append(weight)
            l2=[]
            l2.append(node1)
            l2.append(weight)
            matrix[node1].append(tuple(l1))
            matrix[node2].append(tuple(l2))
        else:
            print(node2,"is not available")
    else:
        print(node1,"is not available")
def show():
    print(matrix)
